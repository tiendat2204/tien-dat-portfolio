## Props

| Prop | Type | Description |
| --- | --- | --- |
| `id` | `string` | Unique identifier for the sticker |
| `frameCount` | `number` | Total number of animation frames |
| `frameRate` | `number` | Speed of animation in frames per second |
| `framesPerCol` | `number` | Number of frames vertically in the sprite sheet |
| `framesPerRow` | `number` | Number of frames horizontally in the sprite sheet |
| `spriteImg` | `string` | URL to the sprite sheet image |
| `className` | `string?` | Optional additional CSS classes |
| `frameSize` | `number?` | Size of each individual frame (default: 64px) |
| `onStickerClick` | `(id: string) => void` | Callback when sticker is clicked |

<CodeBlock
  language="tsx"
  fileName="components/sticker-demo.tsx"
  code={`'use client'
import React, { useState } from 'react'
import StickerCop from '@/components/ui/sticker-cop'

// Sample sticker data
const STICKERS = [
  {
    id: 'thumbs-up',
    frameCount: 20,
    frameRate: 15,
    framesPerRow: 5,
    framesPerCol: 4,
    spriteImg: '/stickers/thumbs-up-sprite.png',
    label: 'Thumbs Up'
  },
  {
    id: 'heart',
    frameCount: 24,
    frameRate: 18,
    framesPerRow: 6,
    framesPerCol: 4,
    spriteImg: '/stickers/heart-sprite.png',
    label: 'Heart'
  },
  // Add more stickers as needed
]

const StickerDemo = () => {
  const [selectedSticker, setSelectedSticker] = useState<string | null>(null)

  const handleStickerClick = (id: string) => {
    setSelectedSticker(id)
    // In a real app, you might send this sticker to a chat
    console.log(\`Selected sticker: \${id}\`)
  }

  return (
    <div className="w-full">
      {selectedSticker && (
        <div className="mb-6 p-3 text-center bg-indigo-50 dark:bg-indigo-900/30 rounded-lg">
          <p className="text-sm font-medium">
            You selected: <span className="text-indigo-600 dark:text-indigo-400">{selectedSticker}</span>
          </p>
        </div>
      )}

      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
        {STICKERS.map((sticker) => (
          <div
            key={sticker.id}
            className={\`p-4 flex flex-col items-center \${
              selectedSticker === sticker.id
                ? 'bg-blue-100 dark:bg-blue-900/40 ring-2 ring-blue-400 dark:ring-blue-500'
                : 'bg-white dark:bg-gray-800'
            } rounded-lg transition-all\`}
          >
            <StickerCop
              id={sticker.id}
              frameCount={sticker.frameCount}
              frameRate={sticker.frameRate}
              framesPerRow={sticker.framesPerRow}
              framesPerCol={sticker.framesPerCol}
              spriteImg={sticker.spriteImg}
              frameSize={80}
              onStickerClick={handleStickerClick}
            />
            <span className="mt-2 text-sm font-medium">{sticker.label}</span>
          </div>
        ))}
      </div>
    </div>
  )
}

export default StickerDemo`}
/>
